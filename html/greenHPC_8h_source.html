258double CPUusage (CPUsage& c, HWconfig& hw) {
259 //this function too can be translated to a lambda since it only does a
260 //single action. Still, first we build the toy model, then we optimize the
261 //structures.
262
263 c.starttime = c.starttime/hw.clock_ticks; //converting to seconds
264 double cpu_occupation = (c.utime + c.stime)/hw.clock_ticks; //converting to seconds
265 double elapsed_time = (c.up_time - c.starttime); //measured in seconds
266 c.elapsed_time = elapsed_time;
267 //std::cout << elapsed_time << '\n';
268 double cpu_usage = cpu_occupation/elapsed_time; //this is actually a percentage
269 return cpu_usage;

301double carbonFootprint (std::vector<double>& cpu_data, std::vector<double>& mem_data, HWconfig& hw, double et) {
302
303 auto n_samples = cpu_data.size();
304
305 auto usage = std::accumulate(std::begin(cpu_data),
306 std::end(cpu_data),
307 0.);
308
309 auto mem_alloc = std::accumulate(std::begin(mem_data),
310 std::end(mem_data),
311 0.);
312
313 auto avg_CPU_usage = usage/n_samples;
314 auto avg_mem_alloc = mem_alloc/n_samples;
315
316 double core_consumption = hw.n_cpu * hw.cpu_tdp * avg_CPU_usage;
317 double mem_consumption = avg_mem_alloc * hw.ram_power_usage;
318
319 return hw.carbon_intensity * (et * (core_consumption + mem_consumption) * hw.pue * 0.001);
320
321}
