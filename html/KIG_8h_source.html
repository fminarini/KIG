264double CPUusage (CPUsage& c, HWconfig& hw) {
265 //this function too can be translated to a lambda since it only does a
266 //single action. Still, first we build the toy model, then we optimize the
267 //structures.
268
269 c.starttime = c.starttime/hw.clock_ticks; //converting to seconds
270 double cpu_occupation = (c.utime + c.stime)/hw.clock_ticks; //converting to seconds
271 double elapsed_time = (c.up_time - c.starttime); //measured in seconds
272 c.elapsed_time = elapsed_time;
273 //std::cout << elapsed_time << '\n';
274 double cpu_usage = cpu_occupation/elapsed_time; //this is actually a percentage
275 return cpu_usage;

307double carbonFootprint (std::vector<double>& cpu_data, std::vector<double>& mem_data, HWconfig& hw, double et) {
308 assert (cpu_data.size() != 0 && mem_data.size() != 0 && et != 0);
309
310 auto n_samples = cpu_data.size();
311
312 auto usage = std::accumulate(std::begin(cpu_data),
313 std::end(cpu_data),
314 0.);
315
316 auto mem_alloc = std::accumulate(std::begin(mem_data),
317 std::end(mem_data),
318 0.);
319
320 auto avg_CPU_usage = usage/n_samples;
321 auto avg_mem_alloc = mem_alloc/n_samples;
322
323 double core_consumption = hw.n_cpu * hw.cpu_tdp * avg_CPU_usage;
324 double mem_consumption = avg_mem_alloc * hw.ram_power_usage;
325
326 return hw.carbon_intensity * (et * (core_consumption + mem_consumption) * hw.pue * 0.001);
327
328}
